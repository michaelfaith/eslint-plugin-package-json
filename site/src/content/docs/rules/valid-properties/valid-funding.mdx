import { Tabs, TabItem } from '@astrojs/starlight/components';

This rule uses [`package-json-validator`](https://github.com/michaelfaith/package-json-validator) to validate the `funding` property, if present, against the following criteria:

- The value is an object, a string, or an Array of objects and strings
- If it's an object, it should have properties `type` and `url`
- `type` should be a non-empty string
- `url` should be a valid URL
- If it's a string, it should be a URL.

## Examples

<Tabs>
<TabItem label="❌ Incorrect">

```json
{
  "funding": {
    "email": "some@mail.com",
    "type": "patreon"
  }
}
```

</TabItem>
<TabItem label="✅ Correct">

```json
{
  "funding": {
    "type": "patreon",
    "url": "https://patreon.com/treznor"
  }
}
```

```json
{
  "funding": "http://npmjs.com/donate"
}
```

```json
{
  "funding": [
    {
      "type": "individual",
      "url": "http://npmjs.com/donate"
    },
    "http://npmjs.com/donate-also",
    {
      "type": "patreon",
      "url": "https://patreon.com/treznor"
    }
  ]
}
```

</TabItem>
</Tabs>
