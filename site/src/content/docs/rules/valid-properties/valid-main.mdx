import { Tabs, TabItem } from '@astrojs/starlight/components';

This rule checks that, if present, the `main` property is a non-empty string.

## Examples

<Tabs>
<TabItem label="❌ Incorrect">

```json
{
  "main": ["index.js", "secondary.js"]
}
```

</TabItem>
<TabItem label="✅ Correct">

```json
{
  "main": "index.js"
}
```

</TabItem>
</Tabs>
