import { TabItem, Tabs } from '@astrojs/starlight/components';

The rule checks that, if present, the `directories` property is an object.

## Examples

<Tabs>
<TabItem label="❌ Incorrect">

```json
{
  "directories": ["bin", "dist"]
}
```

</TabItem>
<TabItem label="✅ Correct">

```json
{
  "directories": {
    "bin": "dist/bin",
    "man": "docs"
  }
}
```

</TabItem>
</Tabs>
