{/* end auto-generated rule header */}

import { Tabs, TabItem } from '@astrojs/starlight/components';

This rule checks for the existence of the `"libc"` property in a package.json, and reports a violation if it doesn't exist.

## Examples

<Tabs>
<TabItem label="❌ Incorrect">

```json
{
  "name": "thee-silver-mt-zion",
  "version": "13.0.0"
}
```

</TabItem>
<TabItem label="✅ Correct">

```json
{
  "name": "thee-silver-mt-zion",
  "version": "13.0.0",
  "libc": ["glibc"]
}
```

</TabItem>
</Tabs>
