import { Aside, TabItem, Tabs } from '@astrojs/starlight/components';

This rule checks for the existence of the `contributors` property and reports a violation if it doesn't exist.

<Aside>
  For publishable packages, consider using the
  [`require-attribution`](/rules/require-attribution) rule instead, which
  requires that _either_ `author` or `contributors` is present.
</Aside>

## Examples

<Tabs>
<TabItem label="❌ Incorrect">

```json
{
  "name": "thee-silver-mt-zion",
  "version": "13.0.0"
}
```

</TabItem>
<TabItem label="✅ Correct">

```json
{
  "name": "thee-silver-mt-zion",
  "version": "13.0.0",
  "contributors": [
    {
      "name": "Trent Reznor",
      "email": "treznor@nin.com",
      "url": "https://nin.com"
    }
  ]
}
```

</TabItem>
</Tabs>
